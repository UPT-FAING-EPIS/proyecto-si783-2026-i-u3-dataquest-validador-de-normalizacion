5def inferir_dependencias_heuristica(tabla): 

6 """ 

7 Infiere dependencias funcionales basadas en patrones de nombres de columnas. 

8 Esto permite al sistema auto-detectar violaciones 2FN y 3FN sin entrada del usuario. 

9 """ 

10 fds = [] 

11 columnas = tabla.get("columnas", []) 

12 pks = tabla.get("pks", []) 

13 

14 # Heurística 1: Dependencias parciales (2FN) 

15 # Si la PK es compuesta, buscar atributos que parezcan pertenecer a solo una parte de la PK. 

16 # Ej: PK(curso_id, profesor_id), atributo: curso_nombre -> depende de curso_id 

17 if len(pks) > 1: 

18 for pk in pks: 

19 pk_base = pk.replace('_id', '').replace('id_', '') 

20 deps_parciales = [] 

21 for col in columnas: 

22 if col not in pks and pk_base in col and col != pk: 

23 deps_parciales.append(col) 

24 if deps_parciales: 

25 fds.append({"determinantes": [pk], "dependientes": deps_parciales}) 

26 

27 # Heurística 2: Dependencias transitivas (3FN) 

28 # Patrón A: columnas con prefijo/sufijo id (ej. departamento_id y departamento_nombre) 

29 for col in columnas: 

30 if col not in pks and (col.endswith('_id') or col.startswith('id_')): 

31 base = col.replace('_id', '').replace('id_', '') 

32 deps_transitivas = [] 

33 for otra_col in columnas: 

34 if otra_col != col and otra_col not in pks and otra_col.startswith(base): 

35 deps_transitivas.append(otra_col) 

36 if deps_transitivas: 

37 fds.append({"determinantes": [col], "dependientes": deps_transitivas}) 

38 

39 # Patrón B: Entidades conceptuales comunes que suelen requerir su propia tabla 

40 entidades_comunes = [ 

41 "departamento", "ciudad", "pais", "marca", "categoria", "proveedor", 

42 "carrera", "facultad", "sucursal", "autor", "editorial", "cliente", 

43 "paciente", "medico", "producto", "estado", "region", "tipo", "rol" 

44 ] 

45 for col in columnas: 

46 if col not in pks: 

47 col_lower = col.lower() 

48 for entidad in entidades_comunes: 

49 if entidad in col_lower: 

50 # Encontramos una posible entidad transitiva. Veamos si hay otra que también la mencione. 

51 deps = [] 

52 for otra_col in columnas: 

53 if otra_col != col and otra_col not in pks and entidad in otra_col.lower(): 

54 deps.append(otra_col) 

55 if deps: 

56 existe = any(d["determinantes"] == [col] for d in fds) 

57 if not existe: 

58 fds.append({"determinantes": [col], "dependientes": deps}) 

59 

60 # Patrón C: Heurística de demasiadas columnas (Tabla "Sábana") 

61 # Si una tabla tiene muchas columnas no-clave (ej. > 6), es muy probable que no esté en 3FN. 

62 # En ese caso, forzamos una dependencia simulada para sugerir separación. 

63 columnas_no_clave = [c for c in columnas if c not in pks] 

64 if len(columnas_no_clave) >= 6: 

65 # Tomar la mitad de las columnas no clave y simular que dependen de una columna inventada o de la primera 

66 det = columnas_no_clave[0] 

67 deps = columnas_no_clave[1:len(columnas_no_clave)//2 + 1] 

68 existe = any(d["determinantes"] == [det] for d in fds) 

69 if not existe: 

70 fds.append({"determinantes": [det], "dependientes": deps}) 

71 

72 return fds

74def diagnosticar_esquema(esquema_parseado): 

75 """ 

76 Analiza un esquema parseado y determina su nivel de normalización actual. 

77 Retorna: 'sin_normalizar', '1fn', '2fn', o '3fn' y un reporte de violaciones. 

78 """ 

79 tablas = esquema_parseado.get("tablas", []) 

80 # Usar las FDs provistas o inferirlas heurísticamente si no hay 

81 fds_globales = esquema_parseado.get("dependencias_funcionales", []) 

82 

83 violaciones_1fn = [] 

84 violaciones_2fn = [] 

85 violaciones_3fn = [] 

86 

87 # Reporte de mejoras opcionales 

88 mejoras_opcionales = [] 

89 

90 nivel_actual = "3FN" # Asumimos optimismo y vamos degradando 

91 

92 for tabla in tablas: 

93 columnas = tabla.get("columnas", []) 

94 

95 # Detectar mejoras opcionales (Semánticas) 

96 for col in columnas: 

97 col_lower = col.lower() 

98 if "nombrecompleto" in col_lower or "nombre_completo" in col_lower: 

99 mejoras_opcionales.append({ 

100 "tabla": tabla["nombre"], 

101 "columna": col, 

102 "tipo": "División Semántica", 

103 "mensaje": f"El atributo '{col}' puede dividirse para mayor flexibilidad.", 

104 "sugerencia_columnas": ["nombre", "apellido_paterno", "apellido_materno"] 

105 }) 

106 elif col_lower == "direccion" or col_lower == "direccion_completa": 

107 mejoras_opcionales.append({ 

108 "tabla": tabla["nombre"], 

109 "columna": col, 

110 "tipo": "División Semántica", 

111 "mensaje": f"El atributo '{col}' puede dividirse para mejorar las búsquedas geográficas.", 

112 "sugerencia_columnas": ["calle", "ciudad", "estado", "codigo_postal"] 

113 }) 

114 

115 # Inferir FDs de la tabla si no se proveyeron globales 

116 fds_tabla = fds_globales if fds_globales else inferir_dependencias_heuristica(tabla) 

117 

118 c1, v1 = validar_1fn(tabla) 

119 violaciones_1fn.extend(v1) 

120 if not c1: 

121 nivel_actual = "sin_normalizar" 

122 

123 c2, v2 = validar_2fn(tabla, fds_tabla) 

124 violaciones_2fn.extend(v2) 

125 if not c2 and nivel_actual in ["3FN", "2FN", "1FN"]: 

126 if nivel_actual != "sin_normalizar": 

127 nivel_actual = "1FN" # Degrada a 1FN 

128 

129 c3, v3 = validar_3fn(tabla, fds_tabla) 

130 violaciones_3fn.extend(v3) 

131 if not c3 and nivel_actual in ["3FN", "2FN"]: 

132 if nivel_actual not in ["sin_normalizar", "1FN"]: 

133 nivel_actual = "2FN" # Degrada a 2FN 

134 

135 # Si llegó aquí y no hay violaciones_3fn, ni de 2fn ni 1fn, es 3FN 

136 

137 reporte = { 

138 "nivel_actual": nivel_actual, 

139 "violaciones_1fn": violaciones_1fn, 

140 "violaciones_2fn": violaciones_2fn, 

141 "violaciones_3fn": violaciones_3fn, 

142 "mejoras_opcionales": mejoras_opcionales, 

143 "resumen": f"El esquema global se encuentra en nivel: {nivel_actual}" 

144 } 

145 

146 return reporte
